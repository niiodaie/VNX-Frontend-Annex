import { useState } from "react";
import { supabase } from "../lib/supabaseClient";

export default function InviteUser() {
  const [email, setEmail] = useState("");
  const [role, setRole] = useState("member");
  const [loading, setLoading] = useState(false);
  const [message, setMessage] = useState("");

  const roles = ["admin", "member", "viewer"];

  async function sendInvite() {
    if (!email) return setMessage("Email is required.");

    setLoading(true);
    setMessage("");

    const { data, error } = await supabase.auth.admin.inviteUserByEmail(email);

    if (error) {
      setMessage("Invite failed: " + error.message);
    } else {
      // Create user profile record in parallel
      await supabase.from("profiles").insert([
        {
          id: data?.user?.id, // This may be undefined immediately
          email,
          role,
        },
      ]);

      setMessage(`Invite sent to ${email} with role ${role}.`);
      setEmail("");
    }

    setLoading(false);
  }

  return (
    <div className="max-w-xl mx-auto mt-10 p-6 bg-white shadow-md rounded">
      <h2 className="text-2xl font-bold mb-4">Invite New User</h2>

      <input
        type="email"
        placeholder="Email address"
        value={email}
        onChange={(e) => setEmail(e.target.value)}
        className="w-full mb-4 p-2 border rounded"
      />

      <select
        value={role}
        onChange={(e) => setRole(e.target.value)}
        className="w-full mb-4 p-2 border rounded"
      >
        {roles.map((r) => (
          <option key={r} value={r}>
            {r.charAt(0).toUpperCase() + r.slice(1)}
          </option>
        ))}
      </select>

      <button
        onClick={sendInvite}
        disabled={loading}
        className="w-full bg-blue-600 text-white py-2 rounded hover:bg-blue-700"
      >
        {loading ? "Sending..." : "Send Invite"}
      </button>

      {message && <p className="mt-4 text-sm text-gray-700">{message}</p>}
    </div>
  );
}

import { useEffect, useState } from "react";
import { supabase } from "../lib/supabaseClient";

export default function DashboardHeader() {
  const [name, setName] = useState("");

  useEffect(() => {
    async function fetchProfile() {
      const {
        data: { user },
      } = await supabase.auth.getUser();

      if (user) {
        const { data: profile } = await supabase
          .from("profiles")
          .select("display_name")
          .eq("id", user.id)
          .single();

        if (profile?.display_name) {
          setName(profile.display_name);
        }
      }
    }

    fetchProfile();
  }, []);

  return (
    <div className="mt-6 mb-4">
      <h1 className="text-2xl font-bold">Welcome back, {name || "there"}!</h1>
      <p className="text-sm text-gray-500">Here's what's happening with your projects today.</p>
    </div>
  );
}
