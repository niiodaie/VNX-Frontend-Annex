// screens/TeenModeScreen.js
import React, { useState } from 'react';
import { View, Text, StyleSheet, TextInput, Button, Alert } from 'react-native';

export default function TeenModeScreen() {
  const [bac, setBac] = useState('');
  const [parentCode, setParentCode] = useState('');

  const handleSubmit = async () => {
    if (parentCode !== '1234') {
      Alert.alert('Invalid Code', 'Please enter the correct parental PIN.');
      return;
    }
    const parsed = parseFloat(bac);
    if (isNaN(parsed)) {
      Alert.alert('Error', 'Enter a valid BAC.');
    } else if (parsed > 0.00) {
      try {
        await fetch('https://your-backend-url.com/api/notify-parent', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ bac: parsed, status: 'fail', teenId: 'teen123' })
        });
        Alert.alert('Warning', 'You should not be driving. Alert sent to parent.');
      } catch (error) {
        Alert.alert('Error', 'Failed to notify parent.');
      }
    } else {
      try {
        await fetch('https://your-backend-url.com/api/notify-parent', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ bac: parsed, status: 'pass', teenId: 'teen123' })
        });
        Alert.alert('Good to Go', 'Drive safe! Parent notified.');
      } catch (error) {
        Alert.alert('Error', 'Failed to notify parent.');
      }
    }
  };

  return (
    <View style={styles.container}>
      <Text style={styles.header}>Teen Driver Check</Text>
      <TextInput
        style={styles.input}
        placeholder="Enter BAC result"
        keyboardType="decimal-pad"
        value={bac}
        onChangeText={setBac}
      />
      <TextInput
        style={styles.input}
        placeholder="Parental PIN"
        secureTextEntry
        keyboardType="number-pad"
        value={parentCode}
        onChangeText={setParentCode}
      />
      <Button title="Submit" onPress={handleSubmit} />
    </View>
  );
}

const styles = StyleSheet.create({
  container: {
    flex: 1,
    padding: 20,
    justifyContent: 'center',
    backgroundColor: '#f0f4f8'
  },
  header: {
    fontSize: 24,
    fontWeight: 'bold',
    marginBottom: 20,
    textAlign: 'center'
  },
  input: {
    height: 50,
    borderColor: '#ccc',
    borderWidth: 1,
    marginBottom: 15,
    paddingHorizontal: 10,
    borderRadius: 8,
    backgroundColor: '#fff'
  }
});
