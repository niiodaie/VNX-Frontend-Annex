// screens/ResultScreen.js (partial logic)
import React, { useState, useEffect } from 'react';
import { View, Text, Button } from 'react-native';
import UpsellModal from '../components/UpsellModal';

export default function ResultScreen({ route }) {
  const [showModal, setShowModal] = useState(false);

  useEffect(() => {
    if (route.params?.result?.upgradeRecommended) {
      setShowModal(true);
    }
  }, [route.params]);

  return (
    <View style={{ flex: 1, justifyContent: 'center', alignItems: 'center' }}>
      <Text>BAC: {route.params?.result?.bac}</Text>
      <Text>Status: {route.params?.result?.message}</Text>
      <UpsellModal visible={showModal} onClose={() => setShowModal(false)} />
    </View>
  );
}
